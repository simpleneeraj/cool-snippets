import React from 'react';
import { BOTTOM_SEGMENT_TABS } from '@/typings/enums';
import { createReducerContext, createSlice } from '@/plugins/mini-state';

// Type for the state of the segment tabs
type SegmentTabsState = {
  selectedBottomTab: BOTTOM_SEGMENT_TABS;
};

// Keys for the state type
type SegmentTabsStateKeys = keyof SegmentTabsState;

// Payload type for actions
type SegmentTabsActionPayload = {
  key: SegmentTabsStateKeys;
  value: any;
};

// Actions type
type SegmentTabsActions = {
  changeSelectedTab: (payload: SegmentTabsActionPayload) => any;
};

// Initial state
const initialState = {
  selectedBottomTab: BOTTOM_SEGMENT_TABS.EDIT,
};

/**
 * Slice Creation
 */
const { reducer, actions } = createSlice<SegmentTabsState, any>(initialState, {
  // Reducer function for handling the changeSelectedTab action
  changeSelectedTab: (state, action) => ({
    ...state,
    [action.payload.key]: action.payload.value,
  }),
});

// Creating a context and provider for the segment tabs state
const [SegmentTabsContext, SegmentTabsProvider] = createReducerContext(
  reducer,
  initialState
);

/**
 * Custom Hook for Tabs
 */
const useSegmentTabs = () => {
  return React.useContext(SegmentTabsContext);
};

// Exporting actions and the context provider
const actionsWithTypeAnnotations = actions as unknown as SegmentTabsActions;
export {
  useSegmentTabs,
  SegmentTabsProvider,
  actionsWithTypeAnnotations as segmentTabsActions,
};
