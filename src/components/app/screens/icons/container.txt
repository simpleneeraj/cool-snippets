import NextImage from 'next/image';
import React, { memo, FC } from 'react';

import { Image } from '@nextui-org/react';
import { emojiCDN } from '@/utils/emoji-cdn';
import UIButton from '@/ui-kit/source/UIButton/button';

type TwemojiProps = {
  emoji: any;
  size?: number;
  skinTone: string;
  onSelectIcon?: (source: string) => void;
};

const TweetEmoji: FC<TwemojiProps> = ({
  emoji,
  size = 24,
  skinTone,
  onSelectIcon,
}) => {
  const source = emojiCDN(emoji, skinTone);

  const onClickButton = React.useCallback(() => {
    onSelectIcon?.(source);
  }, [onSelectIcon, source]);

  return (
    <UIButton isIconOnly variant={'flat'} onPress={onClickButton}>
      <Image
        width={size}
        height={size}
        radius="none"
        // as={NextImage}
        alt={emoji.emoji}
        src={source}
        loading="lazy"
        disableSkeleton
      />
    </UIButton>
  );
};

export default memo(TweetEmoji);
