function moduleLoader(module: () => any) {
  let _module = module();
  let state = {
    loading: true,
    loaded: null,
    error: null,
    module: _module,
  };

  _module
    .then((loaded: null) => {
      state.loading = false;
      state.loaded = resolve(loaded);
      return loaded;
    })
    .catch((err: null) => {
      state.loading = false;
      state.error = err;
      throw err;
    });

  return state;
}

/**************************
Resolve
***************************/
function resolve(obj: any) {
  return obj && obj.__esModule ? obj.default : obj;
}

interface Thenable<T> {
  then<result>(value?: any): PromiseLike<result>;
}

function laz<T>(
  module: () => Thenable<{ default: T }>,
  options: LoadableTypes
) {
  const { fallback, loadingDelay = 100, retryTimeout } = options;

  const payload = {
    _status: null,
    _result: null,
  };

  const loadModule = async () => {
    try {
      let value = (await module().then()) as any;
      await delay(loadingDelay);
      // console.warn(`Resolve module in ${loadingDelay / 1000}s`);
      // console.log(value);
      payload._result = value;
    } catch (e) {
      console.log(e);
    }
  };
  loadModule();
  return payload;
}

export default laz;
